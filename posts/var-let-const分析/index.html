var let const使用和区别

2020-03-04 20:31

var、let 和 const 命令，都是我们最常使用的命令。我们应该如何去使用，且它们有什么区别了。我们通过实际代码去探索。

var 命令

var 声明语句声明一个变量，并可选地将其初始化为一个值。

var a = 1
var b = 2,c = 3 //连续声明
console.log(a) // 输出: 1
console.log(b) // 输出: 2
console.log(c) // 输出: 3

let 命令

let 是 ES6 新增，类似于 var 的功能，但 let 只在作用域中有效

{
  let b = 2
}
console.log(b) // a is not defined (变量a未定义)

const 命令

const 也同样是 ES6 新增命令，声明一个只读的常量。一旦声明，常量的值就不能改变。本质上，const 变量值不能改，只是指 const 声明后的变量的所在内存地址。

const PI = 3.1415
let pi = 3.1415
PI = 3.14 // Assignment to constant variable. (常量重新赋值错误)

pi = 3.14
console.log(pi) // 输出: 3.14

// 声明一个常量c的变量对象
const c = {}
c.name = 'c name' // 允许修改
console.log(c.name) // 输出: c name
c = {} // Assignment to constant variable. (常量重新赋值错误)

var、let 和 const 的区别

变量提升

// var 存在变量提升，即如下
a = 'a string'
var a
console.log(a) // 输出: a string

//可以理解为以下的流程:
var a
a = 'a string'
console.log(a) // 输出: a string

注意: 建议在作用域顶部进行变量声明变量，这样能使得代码更清晰

// let 不存在变量提升
b = 'b string'
let b
console.log(b) // 输出: undefined


// const 不存在变量提升
c = "c string";
const c; // Missing initializer in const declaration (const命令必须先初始化)
console.log(c) // 永不运行到此处

重复声明

var a = 'a string 1'
var a = 'a string 2'
console.log(a) // 输出: a string 2

// let 不允许在同一作用域内重复声明
{
  let b = 'b string 1'
  let b = 'b string 2' // Identifier 'b' has already been declared (b已经被声明)
}

function bFunc(arg) {
  let arg
}
// Identifier 'arg' has already been declared (arg 已经被声明)

//const 不允许重复声明
const c = 'c string 1'
const c = 'c string 2' // Identifier 'b' has already been declared (b已经被声明)

暂时性死区


// var
var a = 'a string 1'
if (true) {
  a = 'a string 2'
  var a
  console.log(a) // 输出: a string 2;
}


// let
var b = 'b string 1'

if (true) {
  b = 'b string 2'
  let b
}
// Cannot access 'a' before initialization (无法在初始化之前访问b)

// const
const c = 'c string 1'
if (true) {
  c = 'c string 2'
  const c
}
// Missing initializer in const declaration (const命令必须先初始化)

顶层对象

{
  var a = 'a string'
  let b = 'b string'
  const c = 'c string'
}
console.log(windows.a) // a string
console.log(windows.b) // undefined
console.log(windows.c) // undefined
